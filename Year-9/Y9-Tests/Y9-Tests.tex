        \item Write \(3.62 \times 10^{-5}\) as an ordinary number. \vspace{3cm}
    \end{enumerate}
    \hfill \textbf{[2 marks]}

    \item
    \begin{enumerate}
        \item Determine the mean, median and mode of the following integers:
        \[
        15, 17, -12, 4, 5, 12, -4, -4.
        \]
        \vspace{6cm}

        \item The mean of 5 numbers is 72. A new number is added and the mean changes to 70. What number was added?

    \vfill \hfill \textbf{[6 marks]}

    \newpage

    \item Fully factorise the following expressions:
    \begin{enumerate}
        \item \(x^2-11x-42\) \vspace{5cm}
        \item \(x^2 - 16x\) \vspace{5cm}
        \item \(4x^2-49\) \vspace{5cm}
        \item \(12-x^2-x\)
    \end{enumerate}
    \vfill \hfill \textbf{[5 marks]}

    \newpage

    \item Calculate the following percentage changes.
    \begin{enumerate}
        \item Increase 56.2 by 13\%. \vspace{4cm}
        \item Decrease 17.4 by 7\%. \vspace{4cm}
        \item £500 is invested into a savings account which accrues 1.3\% compound interest annually. How much money is in the account after 5  years? \vspace{6cm}
        \item Increase 154 by 7\%, then decrease it by 2\% and then increase it by 1\%. What are you left with?
    \end{enumerate}
    \vfill \hfill \textbf{[5 marks]}

    \newpage

    \item Calculate the following without the use of a calculator. You must show all stages in your working. A correct answer with no method will not obtain credit.
    \begin{enumerate}
        \item \(\big(3.2 \times 10^4\big) \div \big(4\times 10^2 \big) \) \vspace{6cm}
        \item \(\big(5.21 \times 10^{-3} \big) - \big( 1.3 \times 10^{-4} \big) \) \vspace{6cm}
    \end{enumerate}
    \vfill \hfill \textbf{[4 marks]}

    \newpage

    \item Solve the following equations.
    \begin{enumerate}
        \item \(x^2+5x-14=0\) \vspace{6cm}
        \item \((3x-1)(x+5)=0\)\vspace{6cm}
        \item \(2(x-1)^2=x(x-7)\)        
    \end{enumerate}
    \vfill \hfill \textbf{[5 marks]}

    \begin{enumerate}
        \item Simplify the following ensuring that your final answers are given in standard form. You may assume that \(p, \, q>0\). \vspace{0.5cm}
        \begin{enumerate}
            \item \(\big(p \times 10^m\big) \times \big(q \times 10^n \big) \) where \(10<pq<100\). \vspace{5cm}
            \item \( \big(p\times10^{2m+1}\big) \div \big(q\times 10^{m-1}\big) \) where \(\frac{q}{100}<p<\frac{q}{10}\). \vspace{5cm}
